\begin{enumerate}
    \item If n is an integer then $n^{2}+3n+5$ is odd.\\
    \quad Case 1: If n is even $\rightarrow$ $n=2k$\\
    \quad \quad $\rightarrow$ $(2k)^{2} + 3(2k) + 5$
        $=> 4k^{2} + 6k + 5$\\
        $=> (4k^{2} + 6k + 4) + 1$\\
        $=> 2(4k^{2} + 3k + 2) + 1$\\
        $=> \boxed{2m + 1}$ \\
    \quad Case 2: If n is odd $\rightarrow$ $n=2k+1$
    \item If $3\mid x$ or $3 \mid y$ then $3 \mid xy$.\\
    \quad Case 1: If some a $\in \mathbb{Z}$, then x = 3a \\
        $\rightarrow xy = (3a)y $\\
        $=> xy = 3(ay)$\\
        $=> \boxed{xy = 3k}$\\
    \quad Case 2: If some b $\in \mathbb{Z}$, then y = 3b \\
    \item If two integers have the same parity, their sum is even.
    \item If two integers have opposite parity, their product is even.
\end{enumerate}
